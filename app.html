import requests

from flask import Flask, render_template


app = Flask(__name__)


@app.route('/')

def home():

    return render_template("home.html")


@app.route('/prices')

def prices():

    url = "https://api.data.gov.in/resource/9ef84268-d588-465a-a308-a864a43d0070?api-key=YOUR_API_KEY&format=json&limit=10"


    response = requests.get(url)

    data = response.json()


    crops = []


    for item in data['records']:

        crop = item.get('commodity', 'N/A')

        price = item.get('modal_price', 'N/A')

        crops.append({"crop": crop, "price": price})


    return render_template("prices.html", data=crops)


@app.route('/sell')

def sell():

    return render_template("sell.html")


@app.route('/login')

def login():

    return render_template("login.html")


if __name__ == '__main__':

    app.run(debug=True)
